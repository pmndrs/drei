Forwards contexts provided above the Canvas to be consumed within it.

### Basic usage

```jsx
function SceneWrapper() {
  const ContextBridge = useContextBridge(ThemeContext, GreetingContext)
  return (
    <Canvas>
      <ContextBridge>
        <Scene />
      </ContextBridge>
    </Canvas>
  )
}
```
